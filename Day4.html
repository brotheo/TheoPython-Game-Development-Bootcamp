üìÑ Full Example Game Structure

import pygame
import random
import sys

pygame.init()
screen = pygame.display.set_mode((800, 600))
pygame.display.set_caption("Mini Game")
clock = pygame.time.Clock()

WHITE = (255, 255, 255)
BLUE = (50, 130, 255)
RED = (255, 50, 50)

player = pygame.Rect(375, 500, 50, 50)
player_speed = 7

enemy = pygame.Rect(random.randint(0, 750), 0, 50, 50)
enemy_speed = 5

font = pygame.font.Font(None, 36)
start_ticks = pygame.time.get_ticks()

game_running = True
while game_running:
    screen.fill(WHITE)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            game_running = False

    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT] and player.left > 0:
        player.x -= player_speed
    if keys[pygame.K_RIGHT] and player.right < 800:
        player.x += player_speed

    enemy.y += enemy_speed
    if enemy.y > 600:
        enemy.y = 0
        enemy.x = random.randint(0, 750)

    if player.colliderect(enemy):
        screen.fill((0, 0, 0))
        game_over = font.render("Game Over", True, RED)
        screen.blit(game_over, (300, 300))
        pygame.display.flip()
        pygame.time.wait(3000)
        game_running = False

    seconds = (pygame.time.get_ticks() - start_ticks) // 1000
    if seconds >= 30:
        win = font.render("You Win!", True, (0, 200, 0))
        screen.blit(win, (320, 280))
        pygame.display.flip()
        pygame.time.wait(3000)
        game_running = False

    pygame.draw.rect(screen, BLUE, player)
    pygame.draw.rect(screen, RED, enemy)
    pygame.display.flip()
    clock.tick(60)

pygame.quit()
sys.exit()
